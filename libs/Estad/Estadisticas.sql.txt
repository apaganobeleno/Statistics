
//tabla temporada
create TABLE Temporada
(
idTemporada INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
fechaIni VARCHAR(200),
fechaFin VARCHAR(200),
nombre VARCHAR(200),
ano VARCHAR(20),
PRIMARY KEY (idTemporada)
)

//tabla torneo
create TABLE Torneo
(
idTorneo INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
fechaIni VARCHAR(200),
nombre VARCHAR(200),
idTemp INTEGER,
pais VARCHAR(200),
PRIMARY KEY (idTorneo)
)

alter table torneo add Foreign Key (idTemp) references Temporada ON delete cascade


//tabla Equipo
create TABLE Equipo(
codigo varchar(20),
nombre varchar(200),
entrenador varchar(200),
asistente varchar(200),
ciudad varchar(200),
PRIMARY KEY(codigo)
)

//tabla Jugador
create table Jugador(
codigo varchar(20),
numero varchar(3),
codigoEquipo varchar(20),
nombre varchar(200),
estatura varchar(10),
peso varchar(10),
PRIMARY KEY(codigo)
)
alter table jugador add Foreign Key (codigoEquipo) references Equipo ON delete cascade

//tabla juego
create table juego(
idJuego INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
idTorneo INTEGER,
eqpPropio VARCHAR(20),
eqpOpp VARCHAR(20),
tipo VARCHAR(20),
fase VARCHAR(20),
dia VARCHAR(20),
juegoNumero VARCHAR(20),
fecha VARCHAR(100),
ciudad VARCHAR(100),
coliseo VARCHAR(100),
espectadores VARCHAR(100),
recaudos VARCHAR(20),
anotador varchar(100),
comopp varchar(100),
compropio varchar(100),
jueces varchar(100),
PRIMARY KEY(idJuego)
)
alter table juego add Foreign Key (idTorneo) references torneo ON delete cascade;
//tabla jugJue
create table jugJue(
codigo VARCHAR(20),
idJuego INTEGER
)
alter table jugJue add Foreign Key (codigo) references jugador ON delete cascade;
alter table jugJue add Foreign Key (idJuego) references juego ON delete cascade;

//tabla tipo
CREATE TABLE Tipo
(
idTipo INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
nombre VARCHAR(20),
letra VARCHAR(2),
PRIMARY KEY (idTipo)
)

//tabla gesto

create TABLE Gesto
(
idGesto INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
nombre VARCHAR(20),
letra VARCHAR(2),
factor INTEGER,
pct INTEGER,
finPos INTEGER,
finNeg INTEGER,
PRIMARY KEY (idGesto)
)

//tabla zona
CREATE TABLE Zona
(
idZona INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
letra VARCHAR(2) NOT NULL,
pos INTEGER,
posFuera INTEGER,
PRIMARY KEY (idZona)
)

//tabla Valor
create TABLE Valor
(
idValor INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
letra VARCHAR(2),
PRIMARY KEY (idValor)
)

//tabla valges- relacion entre valor y gesto, para las tablas de valores configurables.
create TABLE valges
(
idGesto INTEGER,
idValor INTEGER,
valoracion INTEGER,
PRIMARY KEY (idGesto,idValor)
)
alter table valges add Foreign Key (idGesto) references Gesto ON delete cascade;
alter table valges add Foreign Key (idValor) references Valor ON delete cascade;

//Tabla Aceleradores
create Table Acelerador (
idAcelerador INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
descripcion VARCHAR(100),
codigo VARCHAR(100),
Acelerador(100)
PRIMARY KEY(idAcelerador))

//tabla Parcial

create TABLE PARCIAL
(
idParcial INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
idJuego INTEGER,
numero VARCHAR(3),
maximo VARCHAR(3),
posequipos VARCHAR(10),
PRIMARY KEY (idParcial)
)
alter table parcial add Foreign Key (idJuego) references Juego ON delete cascade;

//tabla Rally 
create TABLE RALLY
(
idRally INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
idParcial INTEGER,
ptoPropio VARCHAR(3),
ptoOpp VARCHAR(3),
posP VARCHAR(3),
posAP VARCHAR(3),
BP VARCHAR(2),
aP VARCHAR(20),
P VARCHAR(20),
)
alter table rally add Foreign Key (idParcial) references Parcial ON delete cascade;

//tabla Accion
create TABLE ACCION
(
idAccion INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
codJugador VARCHAR(8),
codEquipo VARCHAR(8),
idJuego INTEGER,
idGesto VARCHAR(8),
idValor VARCHAR(8),
idTipo VARCHAR(8),
idZsalida VARCHAR(8),
idZdestino VARCHAR(8),
idRally INTEGER,
RD VARCHAR(2),
PSO VARCHAR(2),
PRIMARY KEY (idParcial)
)
alter table Accion add Foreign Key (idRally) references rally ON delete cascade;

create TABLE JUGPARCIAL
(
idJugParcial INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
idParcial INTEGER,
CODJUGADOR VARCHAR(20),
INICIAL VARCHAR(2),
ACTUAL VARCHAR(2)
PRIMARY KEY (idJugParcial)
)
alter table JUGPARCIAL add Foreign Key (CODJUGADOR) references jugador ON delete cascade;
alter table JUGPARCIAL add Foreign Key (idParcial) references parcial ON delete cascade;

create table ROTACION(
idRotacion INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1),
numUno varchar(5),
numDos varchar(5),
numTres varchar(5),
numCuatro varchar(5),
numCinco varchar(5),
numSeis varchar(5),
numPas varchar(5),
idParcial INTEGER,
codEqp varchar(100)
)
